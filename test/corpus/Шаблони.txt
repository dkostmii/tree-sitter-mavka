======
Структура із шаблоном
======

структура Яєчко<Т>
  вміст Т
кінець

яєчко_з_числом = Яєчко<число>(вміст=123)

-----


======
Структура із потрійним шаблоном
======

структура Зʼєднувач<Вхід1, Вхід2, Вихід>
  вхід1 Вхід1
  вхід2 Вхід2
  вихід Вихід
кінець

зʼєднувач_списків Зʼєднувач<список<число>, список<число>, список<число> або пусто> = Зʼєднувач<список<число>, список<число>, список<число> або пусто>(
  вхід1=[],
  вхід2=[1, 2, 3],
  вихід=пусто
)

------


======
Дія із шаблоном
======

дія зібрати<Т>(послідовність Т) Агрегат<T>
  ;; зберіть агрегат
кінець

результат = зібрати<список<число>>(послідовність=[1, 2, 3])

-----


======
Дія із потрійним шаблоном
======

дія зʼєднати<Вхід1, Вхід2, Вихід>(вхід1 Вхід1, вхід2 Вхід2) Вихід
  ;; зʼєднайте входи й верніть обʼєкт Виходу
кінець

результат = зʼєднати<список<число>, список<число>>(
  вхід1=[],
  вхід2=[1, 2, 3]
)

------
